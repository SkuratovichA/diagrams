@startuml_seq
@actor user
@participant browser
@participant reseller_ui
user -> browser @actmsgstart Visit the Reseller UI login page
browser -> reseller_ui @actmsgstart Retrieve the Reseller UI login page
browser <- reseller_ui @actmsgstart Return the login page with form field \nusername, password, and One Time Password(OTP)
user <- browser @actmsgstart Display the page, wait for user input @actmsgend
user -> user @actmsgstart Recall username and password \nfrom memory @actmsgend
user -> browser @actmsgstart Fill in the username and password field @actmsgend
user -> user @actmsgstart Open Google Authenticator, \nread the OTP @actmsgend
user -> browser @actmsgstart Fill in the OTP, and hit the send button @actmsgend
browser -> reseller_ui @actmsgstart Send the username, password and OTP @actmsgend
reseller_ui -> reseller_ui @actmsgstart Verify the information is valid @actmsgend
@altstart @condstart Login valid @condend
browser <- reseller_ui @actmsgstart Return the logged in page @actmsgend
user <- browser @actmsgstart Display the logged in page @actmsgend
@else @condstart Login invalid @condend
browser <- reseller_ui @actmsgstart Return login failure page @actmsgend
user <- browser @actmsgstart Display the login failure page @actmsgend
@altend
@enduml_seq
